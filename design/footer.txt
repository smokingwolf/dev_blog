

<script>
window.addEventListener('DOMContentLoaded', () => {
  const links = document.querySelectorAll('a.double-link');

  links.forEach(link => {
    const primaryUrl = link.dataset.primary;
    if (!primaryUrl) return;

    // ★ HTTP→HTTPS に強制変換してテスト
    const httpsPrimary = primaryUrl.replace(/^http:/, 'https:');
    const testUrl = httpsPrimary + (httpsPrimary.includes('?') ? '&' : '?') + 't=' + Date.now();

    // CORSを読めない前提の“存在チェック”：画像プローブでOK/NG判定
    const img = new Image();
    img.onload = () => {
      console.warn('【OK(画像ロード)】hrefを本家(HTTPS)に書換:', httpsPrimary);
      link.href = httpsPrimary;
    };
    img.onerror = () => {
      console.warn('【NG(画像失敗)】アーカイブのまま:', link.href);
      // そのまま
    };
    // Mixed Content回避のため HTTPS のみ叩く
    img.referrerPolicy = 'no-referrer'; // お好み
    img.src = testUrl;
  });
});
</script>



</body>
</html>
